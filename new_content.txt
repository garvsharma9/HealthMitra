        // Modal Functions
        function openApiModal() {
            document.getElementById('sheetIdInput').value = localStorage.getItem('hm_vac_sheet_id') || '';
            document.getElementById('apiKeyInput').value = localStorage.getItem('hm_vac_api_key') || '';

        async function fetchUserData() {
            const sheetId = localStorage.getItem('hm_vac_sheet_id');
            const apiKey = localStorage.getItem('hm_vac_api_key');
            const range = localStorage.getItem('hm_vac_range') || 'Vaccination_Records!A2:I';

            if (!sheetId || !apiKey) {
                setTimeout(() => {
                    originalAppointments = [...dummyAppointments];
                    appointments = [...originalAppointments];
                    sortTable('datetime');
                }, 800);
                return;
            }

            try {
                const url = `https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${range}?key=${apiKey}`;
                const response = await fetch(url);
                if (!response.ok) throw new Error('API Error');
                const result = await response.json();
                
                if (result.values && result.values.length > 0) {
                    originalAppointments = result.values.map(row => ({
                        name: row[0] || 'Unknown',
                        patientId: row[1] || 'N/A',
                        age: row[2] || '',
                        vaccine: row[3] || '',
                        date: row[4] || '',
                        time: row[5] || '',
                        status: row[6] || 'pending',
                        contact: row[7] || '',
                        notes: row[8] || ''
                    }));
                } else {
                    originalAppointments = [];
                }
                appointments = [...originalAppointments];
                sortTable('datetime');
            } catch (err) {
                console.error(err);
                document.getElementById('appointmentsBody').innerHTML = `
                    <tr><td colspan="8" class="no-data"><i class="fas fa-exclamation-triangle" style="color: #f44336;"></i><p>Error connecting to Google Sheets.</p></td></tr>
                `;
            }
        }

        }

        function formatDate(dateStr) {
            if(!dateStr) return '';
            try {
                const date = new Date(dateStr);
                if(isNaN(date)) return dateStr;
                return date.toLocaleDateString('en-IN', {
                    weekday: 'short',
                    day: '2-digit',
                    month: 'short',
                    year: 'numeric'
                });
            } catch(e) { return dateStr; }
        }

        function formatTime(timeStr) {
            if(!timeStr) return '';
            try {
                const parts = timeStr.split(':');
                if(parts.length < 2) return timeStr;
                const date = new Date();
                date.setHours(parts[0], parts[1]);
                return date.toLocaleTimeString('en-IN', {
                    hour: '2-digit',
                    minute: '2-digit',
                    hour12: true
                });
            } catch(e) { return timeStr; }
        }

        function getVaccineClass(vaccine) {
            if(!vaccine) return '';
            return vaccine.toLowerCase().replace(/[^a-z]/g, '-');
        }

        function updateStats() {
            const today = new Date().toISOString().split('T')[0];
            document.getElementById('totalAppointments').textContent = appointments.length;
            document.getElementById('todayAppointments').textContent = appointments.filter(apt => apt.date === today).length;
            document.getElementById('confirmedCount').textContent = appointments.filter(apt => apt.status.toLowerCase() === 'confirmed').length;
            document.getElementById('pendingCount').textContent = appointments.filter(apt => apt.status.toLowerCase() === 'pending').length;
            document.getElementById('completedCount').textContent = appointments.filter(apt => apt.status.toLowerCase() === 'completed').length;
        }

        function sortTable(field) {
            document.querySelectorAll('.sort-btn').forEach(btn => btn.classList.remove('active'));
            if (event && event.target && event.target.classList.contains('sort-btn')) {
                event.target.classList.add('active');
            }

            if (currentSort.field === field) {
                currentSort.direction = currentSort.direction === 'asc' ? 'desc' : 'asc';
            } else {
                currentSort.field = field;
                currentSort.direction = 'asc';
            }

            appointments.sort((a, b) => {
                let valueA, valueB;
                switch(field) {
                    case 'datetime':
                        valueA = new Date(a.date + 'T' + (a.time||'00:00'));
                        valueB = new Date(b.date + 'T' + (b.time||'00:00'));
                        break;
                    case 'date':
                        valueA = new Date(a.date);
                        valueB = new Date(b.date);
                        break;
                    case 'time':
                        valueA = new Date('2025-01-01T' + (a.time||'00:00'));
                        valueB = new Date('2025-01-01T' + (b.time||'00:00'));
                        break;
                    case 'name':
                        valueA = (a.name||'').toLowerCase();
                        valueB = (b.name||'').toLowerCase();
                        break;
                    case 'age':
                        const getMonths = (ageStr) => {
                            if(!ageStr) return 0;
                            const match = ageStr.match(/(\d+)\s*(month|week)/i);
                            if (match) {
                                const num = parseInt(match[1]);
                                return match[2].toLowerCase().startsWith('week') ? num / 4.33 : num;
                            }
                            return 0;
                        };
                        valueA = getMonths(a.age);
                        valueB = getMonths(b.age);
                        break;
                    case 'vaccine':
                        valueA = (a.vaccine||'').toLowerCase();
                        valueB = (b.vaccine||'').toLowerCase();
                        break;
                    case 'status':
                        const statusOrder = { 'confirmed': 1, 'pending': 2, 'completed': 3, 'cancelled': 4 };
                        valueA = statusOrder[(a.status||'').toLowerCase()] || 5;
                        valueB = statusOrder[(b.status||'').toLowerCase()] || 5;
                        break;
                    case 'contact':
                        valueA = (a.contact||'').toLowerCase();
                        valueB = (b.contact||'').toLowerCase();
                        break;
                    default:
                        return 0;
                }

                if (valueA < valueB) return currentSort.direction === 'asc' ? -1 : 1;
                if (valueA > valueB) return currentSort.direction === 'asc' ? 1 : -1;
                return 0;
            });

            renderAppointments();
        }

        function resetTable() {

            document.querySelector('.sort-btn:first-child').classList.add('active');
            sortTable('datetime');
        }

        function searchAppointments() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            if (!searchTerm) {
                appointments = [...originalAppointments];
            } else {
                appointments = originalAppointments.filter(app =>
                    (app.name||'').toLowerCase().includes(searchTerm) ||
                    (app.patientId||'').toLowerCase().includes(searchTerm) ||
                    (app.vaccine||'').toLowerCase().includes(searchTerm) ||
                    (app.status||'').toLowerCase().includes(searchTerm) ||
                    (app.contact||'').includes(searchTerm) ||
                    (app.notes||'').toLowerCase().includes(searchTerm)
                );
            }
            renderAppointments();
        }
